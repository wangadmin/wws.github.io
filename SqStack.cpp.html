#include<stdio.h>

#define MAX_SIZE 5
typedef int ElemType;
typedef struct {
    ElemType data[MAX_SIZE];
    int top;
}SqStack;

//初始化栈
void StackInit(SqStack &stack){
    stack.top = -1; // 此时栈为空
}

//判空
bool StackEmpty(SqStack stack){
    if(stack.top == -1){
        return true;
    }else{
        return false;
    }
}

//入栈
bool Push(SqStack &stack, ElemType e){
    if(stack.top == MAX_SIZE - 1){  //栈满
        return false;
    }
    stack.data[++stack.top] = e;  //先+1后赋值
    return true;
}

//出栈
bool Pop(SqStack &stack, ElemType &e){
    if(StackEmpty(stack)){
        return false;
    }
    e = stack.data[stack.top--];  //先赋值后+1
    return true;
}

//读取栈顶的元素
bool GetTop(SqStack &stack, ElemType &e){
    if(StackEmpty(stack)){
        return false;
    }
    e = stack.data[stack.top];
    return true;
}

int main() {
    SqStack s;
    StackInit(s);
    ElemType e; //存储获取的元素
    bool flag;  //记录操作的结果
    flag = StackEmpty(s);
    if(flag){
        printf("当前栈为空\n");
    }
    Push(s,1);
    Push(s,2);
    Push(s,3);
    Push(s,4);
    Push(s,5);
    flag = Push(s,6);
    if(!flag){
        printf("入栈失败\n");
    }
    flag = Pop(s,e);
    if(flag){
        printf("%d出栈\n",e);
    }
    flag = GetTop(s,e);
    if(flag){
        printf("栈顶元素%d\n",e);
    }
    return 0;
}
