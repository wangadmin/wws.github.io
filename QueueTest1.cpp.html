#include <stdio.h>
#include <stdlib.h>

//2019 42题

typedef int ElemType;
typedef struct LNode{
    ElemType data;
    struct LNode *next;
}LNode;
typedef struct {
    LNode *front,*rear;
}*LiQueue;

void Init(LiQueue &q){
    q = (LiQueue) malloc(sizeof (LiQueue));
    q->front = q->rear = (LNode*) malloc(sizeof (LNode));
    q->front = q->rear;
}

//尾插
void EnQueue(LiQueue &q, ElemType e){
    LNode *p = (LNode*) malloc(sizeof (LNode));
    p->data = e;
    p->next = NULL;
    q->rear->next = p;
    q->rear = p;
}

//头出
bool DeQueue(LiQueue &q, ElemType &e){
    if(q->rear == q->front){
        return false;
    }
    LNode *p = q->front->next;
    q->front->next = p->next;
    e = p->data;
    if(p == q->rear){
        q->front = q->rear;
    }
    free(p);
    return true;
}

int main() {
    LiQueue q;
    Init(q);
    EnQueue(q,1);
    EnQueue(q,2);
    EnQueue(q,3);
    EnQueue(q,4);
    ElemType e;
    DeQueue(q,e);
    printf("%d\n",e);
    DeQueue(q,e);
    printf("%d\n",e);
    return 0;
}
