#include <stdio.h>

#define MAX_SIZE 5
typedef int ElemType;
typedef struct Stack{
    ElemType data[MAX_SIZE];
    int top;
}Stack;
typedef struct Queue{
    ElemType data[MAX_SIZE];
    int front,rear;
}Queue;

void init(Stack &s, Queue &q){
    s.top = -1;
    q.front = q.rear = 0;
}

bool Push(Stack &s, ElemType e){
    if(s.top == MAX_SIZE - 1){
        return false;
    }
    s.data[(++s.top)] = e;
    return true;
}

bool Pop(Stack &s,ElemType &e){
    if(s.top == -1){
        return false;
    }
    e = s.data[(s.top--)];
    return true;
}

bool EnQueue(Queue &q, ElemType e){
    if((q.front+1)%MAX_SIZE == q.rear){
        return false;
    }
    q.data[(q.front++)%MAX_SIZE] = e;
    return true;
}

bool DeQueue(Queue &q, ElemType &e){
    if(q.front == q.rear){
        return false;
    }
    e = q.data[(q.rear++)%MAX_SIZE];
    return true;
}

int main() {
    Stack s;
    Queue q;
    ElemType e;
    bool f;
    int i = 3;
    init(s,q);
    while(i-- > 0){
        scanf("%d",&e);
        Push(s,e);
    }
    i = 3;
    while(i--){
        Pop(s,e);
        printf("%2d",e);
    }
    printf("\n");
    i = 5;
    while(i--){
        scanf("%d",&e);
        f = EnQueue(q,e);
        if(!f){
            printf("false\n");
        }
    }
    i = 4;
    while(i--){
        DeQueue(q,e);
        printf("%2d",e);
    }
    return 0;
}
